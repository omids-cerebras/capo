\subsection{Bayesian graphical model for CAPO}
\label{subsec:capo-plate}

We summarize the \textbf{LV-CAPO} (empirical Bayes) instantiation of CAPO from \S\ref{sec:method} and \S\ref{sec:eb-theory} as a hierarchical Bayesian graphical model. Other CAPO variants correspond to alternative plug-in estimators of the induced variance law $v(L)$ (\S\ref{subsec:capo-plugins}) and plug into the same grouped training loop (\S\ref{sec:algorithms}).

\paragraph{Generative story.}
At a given outer optimization step (suppressing the step index \(t\) for clarity):

\begin{enumerate}
  \item \textbf{Hyperparameters.}  
  Draw global parameters
  \[
    \mu \sim p(\mu)\propto 1,\quad
    \sigma^2 \sim \mathrm{Inv\text{-}Gamma}(a_0,b_0),\quad
    \beta \sim \pi_\beta,\quad
    \xi \sim \pi_\xi,
  \]
  where \(\beta\) controls the length–variance law and \(\xi\) parameterizes the dependence shape \(s(L;\xi)\) (e.g., \(\xi=(\rho,k,\eta)\) under the stretched-geometric $k$-band model of Definition~\ref{def:kband}).
  \item \textbf{Trajectory-level structure.}  

  For each trajectory \(i=1,\dots,G\):
  \begin{enumerate}
    \item The trajectory length \(L_i\in\mathbb N\) is observed (exogenous, determined by prompts and stopping rules).
    \item Conditional on \(\xi\) and \(L_i\), the token-level increments \(\{Y_{i,\tau}\}_{\tau=1}^{L_i}\) form a second–order stationary process with auto-covariance \(\gamma(h;\xi)\). Under the $k$-banded stretched-geometric family,
    \[
      \gamma(0)=\tau^2,\qquad 
      \gamma(h)=\tau^2\rho^{\,|h|^{\eta}}\mathbf 1\{|h|\le k\},
    \]
    so that the induced variance of the scalarized trajectory gradient \(g_i=\sum_{\tau=1}^{L_i} Y_{i,\tau}\) is
    \[
      \mathrm{Var}(g_i\mid L_i,\xi)=\sigma^2 L_i^{\beta} s(L_i;\xi),
      \qquad
      s(L_i;\xi)>0
    \]
    as in Lemma~\ref{lem:sL-kband-eta} and \eqref{eq:hetero_general_prior}.
    \item The scalarized trajectory-gradient statistic \(g_i\in\mathbb R\) obeys the heteroscedastic Normal working model
    \[
      g_i \mid \mu,\sigma^2,\beta,\xi,L_i \sim
      \mathcal N\!\big(\mu,\ v_i(\vartheta)\big),\qquad
      v_i(\vartheta)=\sigma^2 L_i^{\beta} s(L_i;\xi),
    \]
    where \(\vartheta=(\beta,\xi)\).
  \end{enumerate}
\end{enumerate}

\paragraph{Empirical Bayes layer and CAPO aggregator.}
Given observed \(\{(g_i,L_i,\{Y_{i,\tau}\})\}_{i=1}^G\), CAPO performs Empirical Bayes (type-II ML) estimation of \(\vartheta=(\beta,\xi)\) by maximizing the marginal objective \(\ell(\beta,\xi)\) in Theorem~\ref{thm:EB-obj}.  
This induces precision factors
\[
  \omega_i(\beta,\xi)
  = \frac{1}{L_i^{\beta} s(L_i;\xi)},\qquad
  w_i(\beta,\xi)
  = \frac{\omega_i(\beta,\xi)}{\sum_{j=1}^G \omega_j(\beta,\xi)},
\]
and the CAPO aggregator
\[
  m(\beta,\xi) = \sum_{i=1}^G w_i(\beta,\xi)\,g_i,
\]
which is the GLS/Bayes estimator of \(\mu\) for fixed \(\vartheta\) (Proposition~\ref{prop:gls}).  
The weights \(w_i(\beta,\xi)\) are then fed into the CAPO update as trajectory weights for the policy/teacher loss.

\paragraph{Plate diagram.}
Figure~\ref{fig:capo-plate} shows the corresponding Bayesian plate model, highlighting the global hyperparameters \((\mu,\sigma^2,\beta,\xi)\), the trajectory-level variables \((L_i,g_i)\), and the token-level increments \(Y_{i,\tau}\) used to estimate the dependence shape via \(s(L;\xi)\). The deterministic nodes \(v_i(\vartheta)\), \(\omega_i(\beta,\xi)\), and \(m(\beta,\xi)\) summarize the variance law, precision weights, and CAPO aggregator, respectively.
