\section{Proof of the Minimum-Variance Unbiased (MVU) Weights}
\label{app:mvu-weights}

% \section{Inverse-variance MVU weights}
\begin{lemma}[Inverse-variance MVU weights]\label{lem:mvu-weights}
Assume $\mathrm{Cov}(\boldsymbol{g}_i,\boldsymbol{g}_j)=\boldsymbol{0}$ for $i\neq j$ and let $v_i:=\Psi\!\big(\mathrm{Var}(\boldsymbol{g}_i)\big)\in(0,\infty)$ for a fixed positive, linear, monotone functional $\Psi$ on $\mathbb{S}^p_+$ (e.g., $\Psi(\boldsymbol{A})=\boldsymbol{u}^\top\boldsymbol{A}\boldsymbol{u}$ with $\|\boldsymbol{u}\|{=}1$, or $\Psi(\boldsymbol{A})=\mathrm{tr}(\boldsymbol{A})$). Among all linear unbiased estimators $\widehat{\boldsymbol{g}}=\sum_{i=1}^G x_i\boldsymbol{g}_i$ satisfying $\sum_{i=1}^G x_i=1/M$, the weights
\begin{equation}
  x_i^\star \;=\; \frac{1}{M}\,\frac{v_i^{-1}}{\sum_{j=1}^{G} v_j^{-1}}
  \tag{\ref{eq:mvu-weights}}
\end{equation}
uniquely minimize the dispersion $\Psi\!\big(\mathrm{Var}(\widehat{\boldsymbol{g}})\big)$.
\end{lemma}

\begin{proof}
By linearity of covariance and the cross-covariance assumption,
\[
\mathrm{Var}(\widehat{\boldsymbol{g}})=\mathrm{Var}\Big(\sum_{i=1}^G x_i\boldsymbol{g}_i\Big)
=\sum_{i=1}^G x_i^2\,\mathrm{Var}(\boldsymbol{g}_i).
\]
Applying $\Psi$ (positive, linear, monotone on $\mathbb{S}^p_+$),
\[
\Psi\!\big(\mathrm{Var}(\widehat{\boldsymbol{g}})\big)
=\sum_{i=1}^G x_i^2\,\Psi\!\big(\mathrm{Var}(\boldsymbol{g}_i)\big)
=\sum_{i=1}^G v_i\,x_i^2.
\]
Thus the problem reduces to the strictly convex quadratic program
\[
\min_{\boldsymbol{x}}\ \sum_{i=1}^G v_i x_i^2
\quad\text{s.t.}\quad \sum_{i=1}^G x_i=\frac{1}{M},
\]
with $v_i>0$. The Lagrangian is
\(
\mathcal{L}(\boldsymbol{x},\lambda)=\sum_i v_i x_i^2 - \lambda\Big(\sum_i x_i-\frac{1}{M}\Big).
\)
First-order optimality yields \(2v_i x_i - \lambda = 0\), i.e.
\[
x_i=\frac{\lambda}{2v_i}.
\]
Imposing the constraint,
\[
\sum_{i=1}^G x_i=\frac{\lambda}{2}\sum_{i=1}^G \frac{1}{v_i}=\frac{1}{M}
\quad\Longrightarrow\quad
\lambda=\frac{2}{M\sum_{j=1}^G v_j^{-1}}.
\]
Substituting back gives
\[
x_i^\star=\frac{1}{M}\,\frac{v_i^{-1}}{\sum_{j=1}^G v_j^{-1}},
\]
which is the unique minimizer by strict convexity.
\end{proof}

\begin{remark}
Taking $\Psi(\boldsymbol{A})=\boldsymbol{u}^\top\boldsymbol{A}\boldsymbol{u}$ recovers the familiar “inverse-variance” rule with $v_i=\boldsymbol{u}^\top\mathrm{Var}(\boldsymbol{g}_i)\boldsymbol{u}$. Taking $\Psi=\mathrm{tr}$ yields the same weights with $v_i=\mathrm{tr}\,\mathrm{Var}(\boldsymbol{g}_i)$. If $\mathrm{Cov}(\boldsymbol{g}_i,\boldsymbol{g}_j)\neq \boldsymbol{0}$ for some $i\neq j$, the optimal weights depend on the full covariance matrix and are not generally diagonal inverse-variance.
\end{remark}
