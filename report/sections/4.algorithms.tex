\section{Algorithms: modular CAPO with variance-law plug-ins}\label{sec:algorithms}
\label{sec:alg-eb-capo}

\subsection{Workflow overview (precise step-by-step; modular plug-in view)}
CAPO is implemented as a GRPO-style, \emph{grouped} training loop with a modular \emph{variance-law plug-in} (Section~\ref{subsec:capo-plugins}). At each optimizer step, we (i) collect prompt groups, (ii) invoke a plug-in to estimate an induced variance law $\widehat v(L)$ (up to scale) and hence per-trajectory precisions, (iii) normalize these precisions \emph{within each prompt group} to obtain trajectory weights, and (iv) apply the resulting reweighted group objective. LV-CAPO corresponds to the empirical-Bayes plug-in based on \eqref{eq:EB-obj-grouped}; L-CAPO, CAPO-Q, and CAPO-HAC are alternative plug-ins that change only Step~2/4 below.

\begin{enumerate}
  \item \textbf{Rollout collection (prompt groups).} At optimizer step $t$, sample $P_t$ prompts and, for each prompt $p\in\{1,\dots,P_t\}$, sample $N$ trajectories under the current policy $\pi_{\theta_t}$. This yields a grouped batch
  \[
  \mathcal B_t=\big\{(p,i,\,g_{p,i},\,L_{p,i},\,\{Y_{p,i,\tau}\}_{\tau=1}^{L_{p,i}})\big\}_{p=1,i\in\mathcal I_p}^{P_t},
  \]
  where $\mathcal I_p$ indexes the $N$ rollouts in group $p$, $L_{p,i}\in\mathbb N$ is the response length, $g_{p,i}\in\mathbb R$ is a scalarized trajectory-gradient statistic (e.g., a fixed-direction projection of the per-trajectory policy-gradient, \eqref{eq:def_gi_scalar}), and $Y_{p,i,\tau}\in\mathbb R$ are token-level scalar increments available to token-aware plug-ins (LV-CAPO and CAPO-HAC). Outcomes-only plug-ins (L-CAPO, CAPO-Q) depend only on $(g_{p,i},L_{p,i})$.

  \item \textbf{Variance-law plug-in: compute unnormalized precisions.} Choose a plug-in $\mathsf{P}$ that returns a variance-law estimate $\widehat v_{\mathsf{P}}(L)$ (up to scale) and set
  \[
  \widehat\omega_{p,i}\ \propto\ \widehat v_{\mathsf{P}}(L_{p,i})^{-1}.
  \]
  Concretely:
  \begin{itemize}
    \item \textbf{L-CAPO} fixes $s(L)\equiv 1$ and uses $\widehat v(L)=L^{\widehat\beta}$ (with $\widehat\beta$ either fixed to $1$ for $\Delta L$ or estimated online; Algorithm~\ref{alg:eb-lite}).
    \item \textbf{LV-CAPO} uses the structured family $\widehat v(L)=L^{\widehat\beta}s(L;\widehat\xi)$ with $(\widehat\beta,\widehat\xi)$ estimated by grouped EB; Algorithm~\ref{alg:eb-capo}.
    \item \textbf{CAPO-Q} fits a quadratic variance law $\widehat v(L)=\widehat a\,L^{\beta}+\widehat b\,L^{\beta+1}$ motivated by compound symmetry; Algorithm~\ref{alg:capo-q}.
    \item \textbf{CAPO-HAC} estimates the long-run variance by a Bartlett/Newey--West lag window and sets $\widehat v(L)=\widehat v_{\mathrm{HAC}}(L)$; Algorithm~\ref{alg:capo-hac}.
  \end{itemize}

  \item \textbf{Within-group normalization and grouped summaries.} Convert unnormalized precisions into within-group weights
  \[
  \widehat w_{p,i}=\frac{\widehat\omega_{p,i}}{\sum_{j\in\mathcal I_p}\widehat\omega_{p,j}},\qquad
  m_{p,t}=\sum_{i\in\mathcal I_p}\widehat w_{p,i}\,g_{p,i},\qquad
  e_{p,i}=g_{p,i}-m_{p,t}.
  \]
  If the plug-in requires an EB objective (LV-CAPO/L-CAPO), form grouped summaries
  \[
  \Lambda_{\omega,p}=\sum_{i\in\mathcal I_p}\widehat\omega_{p,i},\qquad
  \mathrm{RSS}_{\omega}^{\mathrm{grp}}=\sum_{p=1}^{P_t}\sum_{i\in\mathcal I_p}\widehat\omega_{p,i}\,e_{p,i}^2,
  \]
  matching \eqref{eq:prec-weights-grouped}--\eqref{eq:RSS-grouped}.

  \item \textbf{Plug-in parameter update (optional).} Update plug-in parameters using the current batch:
  \begin{itemize}
    \item \textbf{LV-CAPO}: update $(\beta,\xi)$ by ascent on \eqref{eq:EB-obj-grouped} using \eqref{eq:EB-grad-beta-grouped} and the corresponding $\xi$-gradient (Proposition~\ref{prop:grad} with groupwise means; optionally warm-start via Algorithm~\ref{alg:acf-moment}).
    \item \textbf{L-CAPO}: update $\beta$ only (set $s\equiv 1$) using the same grouped EB structure; $\Delta L$ is the fixed $\beta=1$ special case.
    \item \textbf{CAPO-Q}: update $(a,b)$ by nonnegative least squares on residual squares $e_{p,i}^2$ versus $L_{p,i}^{\beta}$ and $L_{p,i}^{\beta+1}$ (Proposition~\ref{prop:capoq-fit}).
    \item \textbf{CAPO-HAC}: update $\widehat\gamma(h)$ and $\widehat v_{\mathrm{HAC}}(L)$ using lag-window pooling over token increments (Section~\ref{subsec:capo-hac-plugin}).
  \end{itemize}

  \item \textbf{Policy update.} Use the within-group weights $\{\widehat w_{p,i}\}$ to form a weighted GRPO objective (baseline $m_{p,t}$, residual advantages $e_{p,i}$) and take an optimizer step on $\theta$.

  \item \textbf{Stabilization and diagnostics.} Apply EMA to plug-in parameters, enforce box constraints (e.g., $|\rho|\le\rho_{\max}<1$ and $\eta\ge 0$ for $k$-band families), and clip extreme weights. Monitor weight entropy/effective sample size and length-decile stratification.
\end{enumerate}

\subsection{Algorithm~\ref{alg:capo-modular}: Modular CAPO training loop (GRPO prompt groups)}
Algorithm~\ref{alg:capo-modular} makes the modularity explicit: a plug-in produces per-trajectory precisions (equivalently, induced-variance estimates), and the outer loop performs the standard GRPO update using within-group normalized weights.

\begin{algorithm}
\DontPrintSemicolon
\caption{CAPO (modular): GRPO update with a variance-law plug-in}
\label{alg:capo-modular}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Plug-in $\mathsf{P}\in\{\text{L-CAPO},\text{LV-CAPO},\text{CAPO-Q},\text{CAPO-HAC}\}$ with state $\psi$; group size $N$; optimizer and policy $\pi_{\theta}$.}
\Output{Updated policy parameters $\theta$ and updated plug-in state $\psi$ (if applicable).}

\For{$t=1,2,\dots$}{
  Sample $P_t$ prompts; for each prompt $p$ sample $N$ trajectories to form $\mathcal B_t=\{(p,i,g_{p,i},L_{p,i},\{Y_{p,i,\tau}\})\}$.\;
  $(\widehat\omega_{p,i})_{(p,i)\in \mathcal B_t},\ \psi \gets \mathsf{P}.\textsc{FitAndPredict}(\mathcal B_t,\psi)$ \tcp*{plug-in-specific; returns precisions up to scale}\;
  \ForEach{group $p$}{
    $\widehat w_{p,i}\gets \widehat\omega_{p,i}/\sum_{j\in\mathcal I_p}\widehat\omega_{p,j}$;\;
    $m_{p}\gets \sum_{i\in\mathcal I_p}\widehat w_{p,i}g_{p,i}$;\;
    $e_{p,i}\gets g_{p,i}-m_{p}$;\;
  }
  Form the weighted GRPO loss using $(e_{p,i},\widehat w_{p,i})$ and take an optimizer step on $\theta$.\;
  Apply EMA/projection/clipping to plug-in parameters and weights as needed.\;
}
\end{algorithm}

\subsection{Algorithm~\ref{alg:eb-capo}: LV--CAPO in the CAPO training loop}
LV-CAPO is the empirical-Bayes variance-law plug-in that jointly fits a length exponent $\beta$ and a structured dependence shape $\xi$ under the working model of Section~\ref{sec:eb-theory} and its prompt-group extension (Section~\ref{subsec:eb-grouped}). In the modular CAPO loop (Algorithm~\ref{alg:capo-modular}), LV-CAPO is implemented as a \textsc{FitAndPredict} routine that:
\begin{enumerate}
  \item evaluates the current precision weights $\omega_{p,i}(\beta_{t-1},\xi_{t-1})=[L_{p,i}^{\beta_{t-1}}s(L_{p,i};\xi_{t-1})]^{-1}$ on the grouped batch $\mathcal B_t$;
  \item computes within-group weighted means $m_p$ and grouped summaries $(\Lambda_{\omega,p},\mathrm{RSS}^{\mathrm{grp}}_{\omega})$ as in \eqref{eq:prec-weights-grouped}--\eqref{eq:RSS-grouped};
  \item updates $\beta$ (and optionally $\xi$) by ascent on the grouped EB objective \eqref{eq:EB-obj-grouped}, using the grouped length-gradient \eqref{eq:EB-grad-beta-grouped} and either an ACF-moment update (Algorithm~\ref{alg:acf-moment}) or a direct EB gradient update for $\xi$;
  \item returns updated precisions $\omega_{p,i}(\beta_t,\xi_t)$ to be normalized \emph{within each group} by the outer CAPO loop.
\end{enumerate}

\begin{algorithm}
\DontPrintSemicolon
\caption{LV--CAPO plug-in: online grouped-EB estimation and precision prediction}
\label{alg:eb-capo}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Grouped batch $\mathcal B_t=\{(p,i,g_{p,i},L_{p,i},\{Y_{p,i,\tau}\})\}$; previous EB parameters $(\beta_{t-1},\xi_{t-1})$; learning rates $\eta_\beta,\eta_\xi$; EMA coefficients $\gamma_\beta,\gamma_\xi$; constraints/projection $\mathrm{proj}_\Xi$ for $\xi$ (e.g., $|\rho|\le\rho_{\max}<1$, $\eta\ge 0$, fixed $k$).}
\Output{Unnormalized precisions $(\omega_{p,i}^{(t)})_{(p,i)\in\mathcal B_t}$ and updated parameters $(\beta_t,\xi_t)$.}

\tcp{Evaluate current precisions and within-group normalizers}
\ForEach{group $p$}{
  \ForEach{$i\in\mathcal I_p$}{
    $\omega_{p,i}\gets \big(L_{p,i}^{\beta_{t-1}} s(L_{p,i};\xi_{t-1})\big)^{-1}$\;
  }
  $\Lambda_{\omega,p}\gets \sum_{j\in\mathcal I_p}\omega_{p,j}$\;
  \ForEach{$i\in\mathcal I_p$}{
    $w_{p,i}\gets \omega_{p,i}/\Lambda_{\omega,p}$\;
  }
  $m_p\gets \sum_{i\in\mathcal I_p} w_{p,i} g_{p,i}$\;
  \ForEach{$i\in\mathcal I_p$}{
    $e_{p,i}\gets g_{p,i}-m_p$\;
  }
}
$\mathrm{RSS}^{\mathrm{grp}}_{\omega}\gets \sum_{p}\sum_{i\in\mathcal I_p}\omega_{p,i}e_{p,i}^2$\;
$G\gets \sum_{p}|\mathcal I_p|$\;\quad $P\gets \big|\{p:\mathcal I_p\neq\emptyset\}\big|$\;

\tcp{Update $\beta$ by grouped EB ascent (Corollary~\ref{cor:grad-beta-grouped})}
$g_\beta \gets \frac{1}{2}\Big[-\sum_{p}\sum_{i\in\mathcal I_p}\log L_{p,i} + \sum_{p}\frac{\sum_{i\in\mathcal I_p}\omega_{p,i}\log L_{p,i}}{\Lambda_{\omega,p}}\Big]
+ \frac{G-P}{2}\frac{1}{\mathrm{RSS}^{\mathrm{grp}}_{\omega}}\sum_{p}\sum_{i\in\mathcal I_p}\omega_{p,i}e_{p,i}^2\log L_{p,i}
+ \partial_\beta \log \pi_\beta(\beta_{t-1})$\;
$\tilde\beta_t\gets \beta_{t-1}+\eta_\beta g_\beta$\;
$\beta_t\gets (1-\gamma_\beta)\beta_{t-1}+\gamma_\beta\,\mathrm{clip}(\tilde\beta_t;[\beta_{\min},\beta_{\max}])$\;

\tcp{Update $\xi$ (token-aware) either by ACF-moment or EB gradient}
\uIf{ACF-moment update}{
  $\tilde\xi_t\gets \mathrm{ACF\_Moment\_Fit}(\{Y_{p,i,\tau}\},k)$ \tcp*{Algorithm~\ref{alg:acf-moment}}
}
\Else{
  Compute $g_\xi\gets \nabla_\xi \ell_{\mathrm{grp}}(\beta_{t-1},\xi_{t-1})$ (grouped analogue of Proposition~\ref{prop:grad})\;
  $\tilde\xi_t\gets \xi_{t-1}+\eta_\xi g_\xi$\;
}
$\xi_t\gets (1-\gamma_\xi)\xi_{t-1}+\gamma_\xi\,\mathrm{proj}_\Xi(\tilde\xi_t)$\;

\tcp{Return updated precisions for the outer CAPO loop}
\ForEach{group $p$}{
  \ForEach{$i\in\mathcal I_p$}{
    $\omega_{p,i}^{(t)}\gets \big(L_{p,i}^{\beta_t} s(L_{p,i};\xi_t)\big)^{-1}$\;
  }
}
\Return $(\omega_{p,i}^{(t)}),(\beta_t,\xi_t)$\;
\end{algorithm}
\subsection{Algorithm~\ref{alg:acf-moment}: ACF-moment estimator for dependence}
Algorithm~\ref{alg:acf-moment} provides a moment-based plug-in estimator for the dependence shape \(\xi=(\rho,\eta,\dots)\) in the $k$-banded stretched-geometric family \(s(L;\rho,k,\eta)\) (Definition~\ref{def:kband}).
It operates at the \emph{token} level \(Y_{i,\tau}\) and:
\begin{enumerate}
  \item computes per-trajectory empirical autocovariances \(\hat\gamma_i(h)\) up to lag \(k\), centered by the within-trajectory mean \(\bar Y_i\);
  \item forms a pooled autocovariance \(\bar\gamma(h)\) by weighting each trajectory \(i\) by the number of contributing pairs \(L_i-h\);
  \item converts to a pooled autocorrelation \(\bar r(h)=\bar\gamma(h)/\bar\gamma(0)\);
  \item fits \((\rho,\eta)\) by least squares against the theoretical correlation \(\rho^{h^\eta}\) implied by the stretched-geometric model.
\end{enumerate}
The resulting \(\tilde\xi=(\tilde\rho,\tilde\eta,\dots)\) can be used directly in the variance law \(v_i(\vartheta)=\sigma^2 L_i^\beta s(L_i;\xi)\), or as a warm start for subsequent EB gradient updates in Algorithm~\ref{alg:eb-capo} or Algorithm~\ref{alg:joint-eb-kband}.

\begin{algorithm}
\DontPrintSemicolon
\caption{ACF-moment estimator for dependence shape (used inside Algorithm~\ref{alg:eb-capo})}
\label{alg:acf-moment}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Token-level increments \(\{Y_{i,\tau}\}_{\tau=1}^{L_i}\) for each trajectory \(i\); maximum lag \(k\).}
\Output{Shape estimate \(\tilde\xi=(\tilde\rho,\tilde\eta,\dots)\) for the family \(s(L;\rho,k,\eta)\) in Definition~\ref{def:kband}.}

\ForEach{trajectory \(i\)}{
  \(\bar Y_i \gets \dfrac{1}{L_i}\sum_{\tau=1}^{L_i} Y_{i,\tau}\);\;
  \For{\(h=0,1,\dots,k\)}{
    \(\hat\gamma_i(h)\gets \dfrac{1}{L_i-h}\sum_{\tau=1}^{L_i-h} (Y_{i,\tau}-\bar Y_i)(Y_{i,\tau+h}-\bar Y_i)\).\;
  }
}
\(\bar\gamma(h)\gets \dfrac{\sum_i (L_i-h)\hat\gamma_i(h)}{\sum_i (L_i-h)}\) for \(h=0,\dots,k\);\quad
\(\bar r(h)\gets \bar\gamma(h)/\bar\gamma(0)\) for \(h\ge 1\).\;

\(\tilde\xi\gets \arg\min_{\rho,\eta}\sum_{h=1}^k\big(\bar r(h)-\rho^{\,h^{\eta}}\big)^2\)\;
\tcp*{Optionally downweight large \(h\) to reduce noise}
\end{algorithm}

\subsection{Algorithm~\ref{alg:eb-lite}: L-CAPO length-only plug-in (includes $\Delta L$)}
L-CAPO is the \emph{length-only} variance-law plug-in: it fixes the dependence shape to the identity $s(L)\equiv 1$ and models $v(L)\propto L^{\beta}$. In the modular CAPO loop (Algorithm~\ref{alg:capo-modular}), L-CAPO returns unnormalized precisions $\omega_{p,i}\propto L_{p,i}^{-\beta}$ to be normalized within each prompt group.
\begin{itemize}
  \item \textbf{$\Delta L$ as a special case.} Under token independence, $\beta=1$ is correctly specified (Proposition~\ref{prop:lcapo-independence}), yielding $\omega_{p,i}\propto 1/L_{p,i}$.
  \item \textbf{Learned-$\beta$ L-CAPO.} In practice, we can update $\beta$ online via a lightweight moment regression on squared residuals or via a one-dimensional grouped-EB step with $s\equiv 1$.
\end{itemize}

\begin{algorithm}
\DontPrintSemicolon
\caption{L-CAPO plug-in: length-only precisions (optionally fit $\beta$)}
\label{alg:eb-lite}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Grouped batch $\mathcal B_t=\{(p,i,g_{p,i},L_{p,i})\}$; either fixed $\beta$ (e.g., $\beta=1$ for $\Delta L$) or previous estimate $\beta_{t-1}$; small $\varepsilon>0$.}
\Output{Precisions $(\omega_{p,i})$ and (optionally updated) $\beta_t$.}

\uIf{$\beta$ is fixed}{
  \ForEach{group $p$}{
    \ForEach{$i\in\mathcal I_p$}{
      $\omega_{p,i}\gets L_{p,i}^{-\beta}$\;
    }
  }
  \Return $(\omega_{p,i}),\beta$\;
}

\tcp{Compute crude within-group residuals (uniform baseline) and regress log-squared residuals on log-length}
\ForEach{group $p$}{
  $m_p\gets \frac{1}{|\mathcal I_p|}\sum_{i\in\mathcal I_p} g_{p,i}$\;
  \ForEach{$i\in\mathcal I_p$}{
    $e_{p,i}\gets g_{p,i}-m_p$;\quad $z_{p,i}\gets \log(e_{p,i}^2+\varepsilon)$;\quad $\ell_{p,i}\gets \log L_{p,i}$\;
  }
}
Fit $z_{p,i}=c-\beta\,\ell_{p,i}+\epsilon_{p,i}$ by least squares pooled over $(p,i)$; set $\beta_t\gets -$ (estimated slope)\;
\ForEach{group $p$}{
  \ForEach{$i\in\mathcal I_p$}{
    $\omega_{p,i}\gets L_{p,i}^{-\beta_t}$\;
  }
}
\Return $(\omega_{p,i}),\beta_t$\;
\end{algorithm}

\subsection{Algorithm~\ref{alg:capo-q}: CAPO-Q quadratic variance-law plug-in}
CAPO-Q is an outcomes-only plug-in motivated by compound symmetry / random-intercept covariance (Section~\ref{subsec:capo-q-plugin}). It fits a two-term variance law
\[
\widehat v_{\mathrm{Q}}(L)=\widehat a\,L^{\beta}+\widehat b\,L^{\beta+1},
\]
and returns precisions $\omega_{p,i}\propto \widehat v_{\mathrm{Q}}(L_{p,i})^{-1}$ (to be normalized within-group by Algorithm~\ref{alg:capo-modular}). When the compound-symmetry surrogate is correct, CAPO-Q coincides with the oracle inverse-variance rule (Proposition~\ref{prop:capoq-fit}).

\begin{algorithm}
\DontPrintSemicolon
\caption{CAPO-Q plug-in: fit a quadratic induced-variance law and return precisions}
\label{alg:capo-q}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Grouped batch $\mathcal B_t=\{(p,i,g_{p,i},L_{p,i})\}$; fixed $\beta$ (typically $\beta=1$) or an external estimate; nonnegativity constraint $a,b\ge 0$.}
\Output{Precisions $(\omega_{p,i})$ and fitted coefficients $(\widehat a,\widehat b)$.}

\tcp{Compute within-group residuals}
\ForEach{group $p$}{
  $m_p\gets \frac{1}{|\mathcal I_p|}\sum_{i\in\mathcal I_p} g_{p,i}$\;
  \ForEach{$i\in\mathcal I_p$}{
    $e_{p,i}\gets g_{p,i}-m_p$;\quad $y_{p,i}\gets e_{p,i}^2$\;
  }
}
\tcp{Nonnegative least squares fit: $y_{p,i}\approx a L_{p,i}^{\beta}+b L_{p,i}^{\beta+1}$}
$(\widehat a,\widehat b)\gets \arg\min_{a,b\ge 0}\sum_{p,i}\big(y_{p,i}-aL_{p,i}^{\beta}-bL_{p,i}^{\beta+1}\big)^2$\;

\ForEach{group $p$}{
  \ForEach{$i\in\mathcal I_p$}{
    $\widehat v\gets \widehat a\,L_{p,i}^{\beta}+\widehat b\,L_{p,i}^{\beta+1}$;\quad
    $\omega_{p,i}\gets 1/\widehat v$\;
  }
}
\Return $(\omega_{p,i}),(\widehat a,\widehat b)$\;
\end{algorithm}

\subsection{Algorithm~\ref{alg:capo-hac}: CAPO-HAC lag-window long-run variance plug-in}
CAPO-HAC is a token-aware plug-in that estimates the induced variance $v(L)=\mathrm{Var}(\sum_{t=1}^{L}Y_t)$ directly via a Bartlett/Newey--West lag window (Section~\ref{subsec:capo-hac-plugin}). It pools autocovariance estimates across trajectories to reduce noise and returns $\omega_{p,i}\propto \widehat v_{\mathrm{HAC}}(L_{p,i})^{-1}$.

\begin{algorithm}
\DontPrintSemicolon
\caption{CAPO-HAC plug-in: Newey--West induced-variance estimation and precisions}
\label{alg:capo-hac}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Grouped batch $\mathcal B_t=\{(p,i,L_{p,i},\{Y_{p,i,\tau}\})\}$; bandwidth $K\ge 1$.}
\Output{Precisions $(\omega_{p,i})$ and pooled autocovariances $(\widehat\gamma(h))_{h=0}^{K}$.}

\tcp{Estimate pooled autocovariances up to lag $K$}
\For{$h=0,1,\dots,K$}{
  $N_h\gets 0$;\quad $S_h\gets 0$\;
}
\ForEach{trajectory $(p,i)$}{
  $\bar Y_{p,i}\gets \frac{1}{L_{p,i}}\sum_{\tau=1}^{L_{p,i}}Y_{p,i,\tau}$\;
  \For{$h=0,1,\dots,\min\{K,L_{p,i}-1\}$}{
    \For{$\tau=h+1,\dots,L_{p,i}$}{
      $S_h \gets S_h + (Y_{p,i,\tau}-\bar Y_{p,i})(Y_{p,i,\tau-h}-\bar Y_{p,i})$\;
      $N_h \gets N_h + 1$\;
    }
  }
}
\For{$h=0,1,\dots,K$}{
  $\widehat\gamma(h)\gets S_h/N_h$\;
}

\tcp{Map $\widehat\gamma(\cdot)$ to an induced-variance estimate at each length and return precisions}
\ForEach{group $p$}{
  \ForEach{$i\in\mathcal I_p$}{
    $L\gets L_{p,i}$;\quad
    $\widehat v\gets L\,\widehat\gamma(0)$\;
    \For{$h=1,\dots,\min\{K,L-1\}$}{
      $\widehat v\gets \widehat v + 2(L-h)\big(1-\frac{h}{K+1}\big)\widehat\gamma(h)$\;
    }
    $\omega_{p,i}\gets 1/\widehat v$\;
  }
}
\Return $(\omega_{p,i}),(\widehat\gamma(h))_{h=0}^{K}$\;
\end{algorithm}

\subsection{Algorithm~\ref{alg:kband-weights}: \texorpdfstring{$k$}{k}-banded covariance weights}
Algorithm~\ref{alg:kband-weights} specializes the general EB weighting rule to the $k$-banded stretched-geometric dependence model of Definition~\ref{def:kband}.
Given a batch \(\mathcal B_t\), a current length exponent \(\beta_t\), and dependence parameters \((\rho_t,\eta_t)\), it:
\begin{enumerate}
  \item optionally refreshes \((\rho_t,\eta_t)\) using the ACF-moment estimator (Algorithm~\ref{alg:acf-moment});
  \item computes, for each trajectory \(i\), the shape factor
  \[
  s_i = s(L_i;\rho_t,k,\eta_t)
  = 1 + \frac{2}{L_i}\sum_{h=1}^{m_i} (L_i-h)\rho_t^{\,h^{\eta_t}},
  \qquad m_i=\min\{k,L_i-1\};
  \]
  \item forms the precision \(\omega_i^{(t)} = \big(L_i^{\beta_t} s_i\big)^{-1}\) and normalizes to obtain \(w_i^{(t)}\);
  \item returns the dependence-corrected weights \(\{w_i^{(t)}\}\) and the shape factors \(\{s_i\}\).
\end{enumerate}
This gives an explicit implementation of the weighting rule in Proposition~\ref{prop:weights-kband-eta}.

\begin{algorithm}
\DontPrintSemicolon
\caption{k-banded covariance weights (stretched-geometric dependence)}
\label{alg:kband-weights}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Batch \(\mathcal B_t=\{(g_i,L_i,\{Y_{i,\tau}\}_{\tau=1}^{L_i})\}_{i=1}^{G_t}\);\newline
       current length exponent \(\beta_t\);\newline
       current $k$-banded dependence parameters \((\rho_t,\eta_t)\);\newline
       maximum lag \(k\) for the $k$-band family in Definition~\ref{def:kband}.}
\Output{Shape factors \(s_i = s(L_i;\rho_t,k,\eta_t)\) and trajectory weights
        \(w_i^{(t)} \propto [L_i^{\beta_t}\,s(L_i;\rho_t,k,\eta_t)]^{-1}\).}

\tcp{Optional: refresh $(\rho_t,\eta_t)$ from token-level ACFs}
\If{update dependence parameters}{
  \(\tilde\xi_t=(\tilde\rho_t,\tilde\eta_t) \gets
    \mathrm{ACF\_Moment\_Fit}\big(\{Y_{i,\tau}\}_{i\in\mathcal B_t,\tau},k\big)\) \tcp*{Alg.~\ref{alg:acf-moment}}
  \((\rho_t,\eta_t) \gets \mathrm{proj}_\Xi(\tilde\rho_t,\tilde\eta_t)\)\;
}

\tcp{Compute $k$-banded shape factors $s_i$ and precisions $\omega_i$}
\ForEach{\(i\in\mathcal B_t\)}{
  \(m_i \gets \min\{k,\,L_i-1\}\)\;
  \(s_i \gets 1 + \dfrac{2}{L_i}
           \sum_{h=1}^{m_i} (L_i-h)\,\rho_t^{\,h^{\eta_t}}\)
    \tcp*{Definition~\ref{def:kband}}
  \(\omega_i^{(t)} \gets \big(L_i^{\beta_t}\,s_i\big)^{-1}\)\;
}

\tcp{Normalize to obtain Bayes/precision weights}
\(Z_t \gets \sum_{j\in\mathcal B_t} \omega_j^{(t)}\)\;
\ForEach{\(i\in\mathcal B_t\)}{
  \(w_i^{(t)} \gets \omega_i^{(t)} / Z_t\)\;
}

\tcp{Return $k$-banded dependence-corrected weights}
\Return \(\{s_i\}_{i\in\mathcal B_t},\ \{w_i^{(t)}\}_{i\in\mathcal B_t}\)\;
\end{algorithm}

\subsection{Algorithm~\ref{alg:joint-eb-kband}: Joint EB update for \texorpdfstring{$\beta, \rho, \And \eta$}{beta, rho and eta}}
Algorithm~\ref{alg:joint-eb-kband} combines the EB updates for the length exponent \(\beta\) and the $k$-banded dependence parameters \((\rho,\eta)\) into a single joint gradient-ascent step on the EB objective \(\ell(\beta,\rho,\eta)\), under the stretched-geometric dependence model \(s(L;\rho,k,\eta)\).
For a given batch \(\mathcal B_t\), it:
\begin{enumerate}
  \item computes the E-step statistics \(\omega_i, w_i, m_t, e_i, \Lambda_\omega, \mathrm{RSS}_\omega\) under current hyperparameters \((\beta_{t-1},\rho_{t-1},\eta_{t-1})\);
  \item uses Corollary~\ref{cor:grad-beta} to obtain \(g_\beta = \partial_\beta \ell\) and Lemma~\ref{lem:kband-deriv} together with Proposition~\ref{prop:grad} to obtain \(g_\rho = \partial_\rho \ell\) and \(g_\eta = \partial_\eta \ell\);
  \item performs projected gradient-ascent steps on \(\beta,\rho,\eta\), followed by EMA smoothing with \(\gamma_\beta,\gamma_\xi\);
  \item recomputes dependence-corrected weights \(w_i^{(t)}\propto [L_i^{\beta_t}s(L_i;\rho_t,k,\eta_t)]^{-1}\) for use in the CAPO update.
\end{enumerate}
This algorithm is the most direct implementation of the EB theory in \S\ref{sec:eb-theory} when the practitioner is comfortable optimizing the full parameter vector \((\beta,\rho,\eta)\).

\begin{algorithm}
\DontPrintSemicolon
\caption{Joint EB update for length \(\beta\) and \(k\)-banded dependence \((\rho,\eta)\)}
\label{alg:joint-eb-kband}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Batch \(\mathcal B_t=\{(g_i,L_i,\{Y_{i,\tau}\}_{\tau=1}^{L_i})\}_{i=1}^{G_t}\);\;
       current parameters \(\beta_{t-1},\rho_{t-1},\eta_{t-1}\);\;
       learning rates \(\eta_\beta,\eta_\rho,\eta_\eta>0\);\;
       EMA \(\gamma_\beta,\gamma_\xi\in(0,1]\);\;
       constraints \(\beta\in[\beta_{\min},\beta_{\max}],\ |\rho|\le\rho_{\max}<1,\ \eta\ge 0\);\;
       max lag \(k\) for \(s(L;\rho,k,\eta)\) (Def.~\ref{def:kband});\;
       priors \(\pi_\beta,\pi_\rho,\pi_\eta\).}
\Output{Updated \(\beta_t,\rho_t,\eta_t\) and weights
        \(w_i^{(t)} \propto [L_i^{\beta_t}s(L_i;\rho_t,k,\eta_t)]^{-1}\).}

\tcp{Optional warm-start for $(\rho,\eta)$ from ACF moments (Alg.~\ref{alg:acf-moment})}
\If{warm-start}{
  \((\rho_{t-1},\eta_{t-1}) \gets
    \mathrm{proj}_\Xi\big(\mathrm{ACF\_Moment\_Fit}(\{Y_{i,\tau}\},k)\big)\)\;
}

\tcp{E-step: EB statistics under current $(\beta_{t-1},\rho_{t-1},\eta_{t-1})$}
\ForEach{\(i\in\mathcal B_t\)}{
  \(m_i \gets \min\{k,\,L_i-1\}\);\;
  \(s_i \gets 1 + \dfrac{2}{L_i}\sum_{h=1}^{m_i} (L_i-h)\,\rho_{t-1}^{\,h^{\eta_{t-1}}}\);\;
  \(\omega_i \gets \big(L_i^{\beta_{t-1}}\,s_i\big)^{-1}\)\;
}
\(\Lambda_\omega \gets \sum_{i\in\mathcal B_t} \omega_i\);\;
\ForEach{\(i\in\mathcal B_t\)}{
  \(w_i \gets \omega_i / \Lambda_\omega\)\;
}
\(m_t \gets \sum_{i\in\mathcal B_t} w_i g_i\);\;
\ForEach{\(i\in\mathcal B_t\)}{
  \(e_i \gets g_i - m_t\)\;
}
\(\mathrm{RSS}_\omega \gets \sum_{i\in\mathcal B_t} \omega_i e_i^2\)\;

\tcp{M-step: gradients of EB objective $\ell(\beta,\rho,\eta)$}
Compute \(g_\beta = \partial_\beta \ell\) from Cor.~\ref{cor:grad-beta}
(using current \(\{\omega_i,e_i,L_i\}\))\;
Compute \(g_\rho = \partial_\rho \ell\) and \(g_\eta = \partial_\eta \ell\)
from Prop.~\ref{prop:grad} and Lem.~\ref{lem:kband-deriv}
(using current \(\{\omega_i,e_i,s_i\}\))\;

\tcp{Gradient-ascent updates with projection and EMA}
\(\tilde\beta_t \gets \beta_{t-1} + \eta_\beta\, g_\beta\);\(\tilde\rho_t \gets \rho_{t-1} + \eta_\rho\, g_\rho\);\(\tilde\eta_t \gets \eta_{t-1} + \eta_\eta\, g_\eta\);\;

\(\beta_t \gets (1-\gamma_\beta)\beta_{t-1}
  + \gamma_\beta\,\mathrm{clip}(\tilde\beta_t;[\beta_{\min},\beta_{\max}])\);\;
\(\rho_t \gets (1-\gamma_\xi)\rho_{t-1}
  + \gamma_\xi\,\mathrm{clip}(\tilde\rho_t;[-\rho_{\max},\rho_{\max}])\);\;
\(\eta_t \gets (1-\gamma_\xi)\eta_{t-1}
  + \gamma_\xi\,\max\{\tilde\eta_t,\,0\}\);\;

\tcp{Recompute $k$-banded weights at $(\beta_t,\rho_t,\eta_t)$ for CAPO}
\ForEach{\(i\in\mathcal B_t\)}{
  \(m_i \gets \min\{k,\,L_i-1\}\);\;
  \(s_i \gets 1 + \dfrac{2}{L_i}\sum_{h=1}^{m_i} (L_i-h)\,\rho_t^{\,h^{\eta_t}}\);\;
  \(\omega_i \gets \big(L_i^{\beta_t}\,s_i\big)^{-1}\)\;
}
\(\Lambda_\omega \gets \sum_{j\in\mathcal B_t} \omega_j\);\;
\ForEach{\(i\in\mathcal B_t\)}{
  \(w_i^{(t)} \gets \omega_i / \Lambda_\omega\)\;
}

\Return \(\beta_t,\rho_t,\eta_t,\{w_i^{(t)}\}_{i\in\mathcal B_t}\)\;
\end{algorithm}

\paragraph{Complexity and implementation notes.}
Per outer step \(t\), computing the EB statistics \(\{\omega_i^{(t)},w_i^{(t)},m_t,e_i^{(t)},\Lambda_\omega^{(t)},\mathrm{RSS}_\omega^{(t)}\}\) has cost \(O(G_t)\) for a batch of \(G_t\) trajectories.
The ACF-moment estimator with maximum lag \(k\) has cost \(O\big(\sum_{i\in\mathcal B_t} L_i\,k\big)\), which is typically dominated by the rollout generation cost.
In practice, small EMA coefficients \(\gamma_\beta,\gamma_\xi\in[0.05,0.3]\), a bound \(|\rho|\le 0.9\), and clipping of the largest \(w_i^{(t)}\) (e.g., at the 99.5th percentile) are sufficient to prevent any single trajectory from dominating the update.

\paragraph{Numerical stability, identifiability, and practical defaults.}
In RLVR regimes with long-context CoT, the raw length factors $L_i^{-\beta}$ can underflow for large $L_i$ and moderate $\beta$.
A numerically stable implementation should therefore compute weights in log-space:
$\log \omega_i = -\beta \log L_i - \log s(L_i;\xi)$ and then use a log-sum-exp reduction to obtain $\Lambda_\omega=\sum_i \omega_i$.
Similarly, when the dependence model includes exponential terms (e.g., $e^{-L/\xi}$), it is preferable to evaluate $\log s(L;\xi)$ directly.

From an identifiability perspective, $(\beta,\xi)$ are only weakly identifiable in small batches when the empirical length distribution is narrow.
In such cases, we recommend (i) restricting $\beta$ to a conservative interval (e.g., $[0,1]$), (ii) using the outcomes-only EB objective as a warm-start for the full LV-CAPO updates, and (iii) employing mild regularization or informative priors as in \S\ref{subsec:informative-beta}.
These steps stabilize estimation without altering the unbiasedness constraint in \eqref{eq:linear-agg}.
